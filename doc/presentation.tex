\section{Diet Optimization}

\begin{frame}
\frametitle{Multi-Objective Optimization}
\begin{block}{Objective Function}
\[ S = w_n S_n + w_t S_t + w_d S_d + w_v S_v \]
\end{block}

\begin{itemize}
    \item $S_n$: Nutrient compliance score
    \item $S_t$: Taste preference score
    \item $S_d$: Digestion quality score
    \item $S_v$: Food variety score
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Optimization Algorithm}
\begin{enumerate}
    \item \textbf{Initialization}: Random amounts of seasonal foods
    \item \textbf{Scoring}: Calculate fitness based on all objectives
    \item \textbf{Optimization}: Iteratively adjust food amounts
    \item \textbf{Meal Distribution}: Distribute across breakfast, lunch, dinner
    \item \textbf{Validation}: Ensure USDA compliance
\end{enumerate}
\end{frame}
