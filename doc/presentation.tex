\begin{itemize}
    \item \textbf{Sunday}: Music (30 min) + Hearing Exercise (10 min)
    \item \textbf{Monday}: Hearing Exercise (10 min)
    \item \textbf{Tuesday}: Active Music (20 min)
    \item \textbf{Thursday}: Music Session (30 min)
    \item \textbf{Friday}: Binaural Beats (20 min)

    \item Detect hearing loss → Suggest targeted exercises
    \item Identify tinnitus → Recommend therapy activities
\end{enumerate}
